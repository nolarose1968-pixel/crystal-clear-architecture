// Crystal Clear Architecture Example interface DomainEntity { id: string; createdAt: Date; updatedAt: Date; } interface FinancialReport extends DomainEntity { period: string; revenue: number; expenses: number; profit: number; compliance: boolean; } class FinancialReportingService { private reports: FinancialReport[] = []; async generateReport(period: string): Promise { // Simulate report generation const revenue = Math.random() * 100000; const expenses = revenue * 0.7; const profit = revenue - expenses; const report: FinancialReport = { id: `report-${Date.now()}`, period, revenue: Math.round(revenue), expenses: Math.round(expenses), profit: Math.round(profit), compliance: Math.random() > 0.1, // 90% compliance rate createdAt: new Date(), updatedAt: new Date() }; this.reports.push(report); return report; } getAllReports(): FinancialReport[] { return this.reports; } getComplianceRate(): number { if (this.reports.length === 0) return 100; const compliant = this.reports.filter(r => r.compliance).length; return Math.round((compliant / this.reports.length) * 100); } } // Usage example const reportingService = new FinancialReportingService(); async function demonstrateReporting() { console.log('🧪 Crystal Clear Financial Reporting Demo'); console.log('=' .repeat(50)); // Generate sample reports const periods = ['Q1-2024', 'Q2-2024', 'Q3-2024', 'Q4-2024']; for (const period of periods) { const report = await reportingService.generateReport(period); console.log(`📊 ${period} Report:`); console.log(` Revenue: $${report.revenue.toLocaleString()}`); console.log(` Expenses: $${report.expenses.toLocaleString()}`); console.log(` Profit: $${report.profit.toLocaleString()}`); console.log(` Compliance: ${report.compliance ? '✅' : '❌'}`); console.log(''); } const allReports = reportingService.getAllReports(); const totalRevenue = allReports.reduce((sum, r) => sum + r.revenue, 0); const totalProfit = allReports.reduce((sum, r) => sum + r.profit, 0); const complianceRate = reportingService.getComplianceRate(); console.log('📈 Summary Statistics:'); console.log(` Total Reports: ${allReports.length}`); console.log(` Total Revenue: $${totalRevenue.toLocaleString()}`); console.log(` Total Profit: $${totalProfit.toLocaleString()}`); console.log(` Compliance Rate: ${complianceRate}%`); console.log(` Average Profit Margin: ${((totalProfit / totalRevenue) * 100).toFixed(1)}%`); } demonstrateReporting();" data-language="typescript"> 🏗️
