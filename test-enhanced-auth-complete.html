🔧 Advanced Configuration

// Custom rate limiting const api = new APIService('https://api.example.com'); api.rateLimiter = new RateLimiter(200, 60000); // 200 requests per minute // Custom caching api.cacheManager = new CacheManager(10 * 60 * 1000); // 10 minutes TTL // Custom security headers api.defaultHeaders = { ...api.defaultHeaders, 'X-Custom-Header': 'custom-value', 'X-Request-ID': generateRequestId() };

🚨 Error Handling

// Comprehensive error handling try { const result = await api.request('/api/v1/protected'); return result; } catch (error) { if (error.message.includes('Rate limit exceeded')) { // Handle rate limiting showRateLimitError(); } else if (error.message.includes('Session expired')) { // Handle authentication error redirectToLogin(); } else if (error.message.includes('Network error')) { // Handle network issues showNetworkError(); } else { // Handle other errors showGenericError(error.message); } }
