üì± User Registration & Linking

How users connect their Telegram accounts to your dashboard:

# Bot Start Command - User Registration
bot.onText(/\/start/, async (msg) => {
  const telegramId = msg.from.id;
  const username = msg.from.username;
  
  try {
    // Check if user exists
    const existingUser = await db.get(
      'SELECT * FROM users WHERE telegram_id = ?', 
      [telegramId]
    );
    
    if (!existingUser) {
      // New user - create account
      await db.run(`
        INSERT INTO users (name, telegram_id, telegram_username) 
        VALUES (?, ?, ?)
      `, [username, telegramId, username]);
      
      bot.sendMessage(telegramId, 
        'üéâ Welcome to Fire22 Dashboard! Your account has been created.');
    } else {
      bot.sendMessage(telegramId, 
        'üëã Welcome back! Use /dashboard to access your account.');
    }
  } catch (error) {
    console.error('Registration error:', error);
    bot.sendMessage(telegramId, '‚ùå Registration failed. Please try again.');
  }
});

# Link Existing Account
bot.onText(/\/link (.+)/, async (msg, match) => {
  const telegramId = msg.from.id;
  const email = match[1];
  
  try {
    await db.run(`
      UPDATE users 
      SET telegram_id = ?, telegram_linked_at = CURRENT_TIMESTAMP
      WHERE email = ?
    `, [telegramId, email]);
    
    bot.sendMessage(telegramId, '‚úÖ Account linked successfully!');
  } catch (error) {
    bot.sendMessage(telegramId, '‚ùå Failed to link account. Please check your email.');
  }
});

üîí Security & Validation

Secure your Telegram integration:

# Telegram ID Validation
const validateTelegramId = (telegramId: number): boolean => {
  const idString = telegramId.toString();
  return idString.length >= 9 && idString.length <= 10 && 
         /^\d+$/.test(idString);
};

# Access Control
const getUserData = async (requestedUserId: number, telegramId: number) => {
  const user = await db.get(`
    SELECT * FROM users 
    WHERE id = ? AND telegram_id = ?
  `, [requestedUserId, telegramId]);
  
  if (!user) {
    throw new Error('Access denied: User not found or unauthorized');
  }
  
  return user;
};

# Group Creation & Management
const createSupportGroup = async (groupData: any) => {
  const { name, description, telegramGroupId, serviceLevel } = groupData;
  
  try {
    const result = await db.run(`
      INSERT INTO support_groups (name, description, telegram_group_id, service_level)
      VALUES (?, ?, ?, ?)
    `, [name, description, telegramGroupId, serviceLevel]);
    
    // Create Telegram group if it doesn't exist

    return { success: true, groupId: result.lastID };
  } catch (error) {
    console.error('Failed to create support group:', error);
    return { success: false, error: error.message };
  }
};

# Add User to Group
const addUserToGroup = async (userId: number, groupId: number, role: string = 'member') => {
  try {
    const user = await db.get('SELECT telegram_id FROM users WHERE id = ?', [userId]);
    const group = await db.get('SELECT telegram_group_id FROM support_groups WHERE id = ?', [groupId]);
    
    if (user?.telegram_id && group?.telegram_group_id) {
      await db.run(`
        INSERT OR REPLACE INTO user_groups (user_id, group_id, role)
        VALUES (?, ?, ?)
      `, [userId, groupId, role]);
      
      // Send welcome message to group
      await bot.sendMessage(group.telegram_group_id, 
        `üëã New member joined: @${user.telegram_username || 'User'}`);
      
      return { success: true };
    }
  } catch (error) {
    console.error('Failed to add user to group:', error);
    return { success: false, error: error.message };
  }
};

üéØ Customer Service Levels & Priority Management

Implement tiered customer service with automated escalation and SLA management:

# Service Level Configuration
CREATE TABLE service_levels (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL, -- basic, premium, vip, enterprise
  response_time_minutes INTEGER NOT NULL,
  escalation_time_minutes INTEGER,
  auto_escalate BOOLEAN DEFAULT true,
  priority_score INTEGER DEFAULT 1,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE support_tickets (
  id INTEGER PRIMARY KEY,
  user_id INTEGER NOT NULL,
  group_id INTEGER,
  service_level TEXT NOT NULL,
  priority TEXT DEFAULT 'normal',
  status TEXT DEFAULT 'open',
  subject TEXT NOT NULL,
  message TEXT NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  first_response_at TIMESTAMP,
  resolved_at TIMESTAMP,
  FOREIGN KEY(user_id) REFERENCES users(id),
  FOREIGN KEY(group_id) REFERENCES support_groups(id)
);
# Service Level Management
const serviceLevels = {
  basic: { responseTime: 240, escalationTime: 480, priority: 1 },      // 4h response, 8h escalation
  premium: { responseTime: 120, escalationTime: 240, priority: 2 },    // 2h response, 4h escalation
  vip: { responseTime: 60, escalationTime: 120, priority: 3 },         // 1h response, 2h escalation
  enterprise: { responseTime: 30, escalationTime: 60, priority: 4 }    // 30m response, 1h escalation
};

# Create Support Ticket with Priority
const createSupportTicket = async (ticketData: any) => {
  const { userId, subject, message, serviceLevel = 'basic' } = ticketData;
  
  try {
    const level = serviceLevels[serviceLevel];
    const priority = level.priority >= 3 ? 'high' : 'normal';
    
    const result = await db.run(`
      INSERT INTO support_tickets (user_id, service_level, priority, subject, message)
      VALUES (?, ?, ?, ?, ?)
    `, [userId, serviceLevel, priority, subject, message]);
    
    const ticketId = result.lastID;
    
    // Send immediate notification to appropriate support group
    await notifySupportTeam(ticketId, serviceLevel, priority);
    
    // Schedule escalation check
    setTimeout(() => checkTicketEscalation(ticketId), level.escalationTime * 60 * 1000);
    
    return { success: true, ticketId };
  } catch (error) {
    console.error('Failed to create support ticket:', error);
    return { success: false, error: error.message };
  }
};
# Automated Escalation System
const checkTicketEscalation = async (ticketId: number) => {
  try {
    const ticket = await db.get(`
      SELECT t.*, u.telegram_id, u.name 
      FROM support_tickets t 
      JOIN users u ON t.user_id = u.id 
      WHERE t.id = ? AND t.status = 'open'
    `, [ticketId]);
    
    if (!ticket) return;
    
    const level = serviceLevels[ticket.service_level];
    const timeSinceCreation = Date.now() - new Date(ticket.created_at).getTime();
    const responseTimeMs = level.responseTime * 60 * 1000;
    
    if (timeSinceCreation > responseTimeMs && !ticket.first_response_at) {
      // Escalate ticket
      await escalateTicket(ticketId, ticket);
      
      // Send urgent notification to support team
      await sendEscalationAlert(ticket);
      
      // Notify user about escalation
      if (ticket.telegram_id) {
        await bot.sendMessage(ticket.telegram_id, 
          `üö® Your support ticket has been escalated due to response time.\nTicket: ${ticket.subject}\nWe apologize for the delay and are working on it.`);
      }
    }
  } catch (error) {
    console.error('Escalation check failed:', error);
  }
};

const escalateTicket = async (ticketId: number, ticket: any) => {
  await db.run(`
    UPDATE support_tickets 
    SET priority = 'urgent', status = 'escalated'
    WHERE id = ?
  `, [ticketId]);
  
  // Find appropriate escalation group
  const escalationGroup = await db.get(`
    SELECT * FROM support_groups 
    WHERE service_level = ? AND name LIKE '%escalation%'
    LIMIT 1
  `, [ticket.service_level]);
  
  if (escalationGroup) {
    await db.run(`
      UPDATE support_tickets SET group_id = ? WHERE id = ?
    `, [escalationGroup.id, ticketId]);
  }
};
# SLA Monitoring & Reporting
const generateSLAReport = async (timeframe: string = 'month') => {
  try {
    const report = await db.all(`
      SELECT 
        service_level,
        COUNT(*) as total_tickets,
        AVG(CASE WHEN first_response_at IS NOT NULL 
          THEN (julianday(first_response_at) - julianday(created_at)) * 1440 
          ELSE NULL END) as avg_response_time_minutes,
        COUNT(CASE WHEN status = 'resolved' THEN 1 END) as resolved_tickets,
        COUNT(CASE WHEN status = 'escalated' THEN 1 END) as escalated_tickets
      FROM support_tickets 
      WHERE created_at >= datetime('now', '-1 ${timeframe}')
      GROUP BY service_level
      ORDER BY service_level
    `);
    
    return report.map(level => ({
      ...level,
      sla_compliance: calculateSLACompliance(level),
      response_time_grade: gradeResponseTime(level.avg_response_time_minutes)
    }));
  } catch (error) {
    console.error('SLA report generation failed:', error);
    return [];
  }
};

const calculateSLACompliance = (level: any) => {
  const target = serviceLevels[level.service_level]?.responseTime || 240;
  const actual = level.avg_response_time_minutes || 0;
  return actual <= target ? '‚úÖ Compliant' : '‚ùå Non-compliant';
};
