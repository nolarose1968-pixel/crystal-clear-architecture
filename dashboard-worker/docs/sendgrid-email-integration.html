üöÄ Quick Start

Get started with SendGrid email integration in just a few steps:

# 1. Install SendGrid SDK
bun add @sendgrid/mail

# 2. Set environment variables
bun run env:set SENDGRID_API_KEY=SG.your_api_key_here

bun run env:integration

üîß Template Management

Create and manage email templates programmatically:

# Create Dynamic Template
const createTemplate = async (templateData) => {
  const response = await sgMail.request({
    method: 'POST',
    url: '/v3/templates',
    body: {
      name: templateData.name,
      generation: 'dynamic',
      versions: [{
        name: 'version_1',
        subject: templateData.subject,
        html_content: templateData.htmlContent,
        plain_content: templateData.plainContent
      }]
    }
  });
  
  return response.body;
};

# Send with Template
const sendTemplateEmail = async (to, templateId, data) => {
  const msg = {
    to,
    from: config.sendgrid.fromEmail,
    templateId,
    dynamicTemplateData: data
  };
  
  try {
    const result = await sgMail.send(msg);
    console.log('Email sent successfully', result);
    return result;
  } catch (error) {
    console.error('Email sending failed', error);
    throw error;
  }
};

üìä Email Analytics & Tracking

Monitor email performance and delivery metrics:

# Get Email Statistics
const getEmailStats = async (startDate, endDate) => {
  const response = await sgMail.request({
    method: 'GET',
    url: '/v3/stats',
    qs: {
      start_date: startDate,
      end_date: endDate,
      aggregated_by: 'day'
    }
  });
  
  return response.body;
};

# Track Email Events
const setupWebhook = '/webhook/sendgrid';
const handleWebhook = (req, res) => {
  const events = req.body;
  
  events.forEach(event => {
    switch (event.event) {
      case 'delivered':
        console.log(`Email delivered: ${event.email}`);
        break;
      case 'opened':
        console.log(`Email opened: ${event.email}`);
        break;
      case 'clicked':
        console.log(`Link clicked: ${event.email}`);
        break;
      case 'bounced':
        console.log(`Email bounced: ${event.email}`);
        break;
    }
  });
  
  res.sendStatus(200);
};

üîí Security & Best Practices

Secure your SendGrid integration:

# API Key Security
const validateApiKey = (apiKey) => {
  if (!apiKey || !apiKey.startsWith('SG.')) {
    throw new Error('Invalid SendGrid API key format');
  }
  
  if (apiKey.length < 50) {
    throw new Error('API key too short');
  }
  
  return true;
};

# Rate Limiting
class SendGridRateLimiter {
  constructor(maxEmailsPerSecond = 100) {
    this.maxEmailsPerSecond = maxEmailsPerSecond;
    this.emailQueue = [];
    this.lastSent = 0;
  }
  
  async sendEmail(emailData) {
    const now = Date.now();
    const timeSinceLastSent = now - this.lastSent;
    
    if (timeSinceLastSent < (1000 / this.maxEmailsPerSecond)) {
      await new Promise(resolve => 
        setTimeout(resolve, (1000 / this.maxEmailsPerSecond) - timeSinceLastSent)
      );
    }
    
    this.lastSent = Date.now();
    return await sgMail.send(emailData);
  }
}
