🔧 Technical Implementation

How to integrate Fire22's multilingual Telegram system:

# Core Multilingual Bot Integration
import { Fire22TelegramIntegration } from './src/telegram/fire22-telegram-integration';
import { MultilingualTelegramBot } from './src/telegram/multilingual-telegram-bot';

# Initialize multilingual bot
const config = {
  botToken: process.env.BOT_TOKEN,
  cashierGroupId: process.env.CASHIER_GROUP_ID,
  p2pQueueGroupId: process.env.P2P_QUEUE_GROUP_ID,
  supportGroupId: process.env.SUPPORT_GROUP_ID
};

const telegramBot = new Fire22TelegramIntegration(config, telegramAPI);

# Send multilingual notifications
await telegramBot.sendTransactionNotification(userId, {
  amount: 500,
  type: 'Deposit',
  timestamp: new Date()
});

# Language detection and switching
const multiBot = new MultilingualTelegramBot();
const userLang = multiBot.getUserLanguage(telegramUser);
multiBot.setUserLanguage(userId, 'es'); // Switch to Spanish
