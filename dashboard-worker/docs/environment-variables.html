Get started with environment variables in just a few steps:

1. Copy the example environment file:
cp .env.example .env
2. Fill in your environment variables in .env
3. Validate your configuration:
bun run env:validate

📁 Environment File Structure

Bun automatically loads environment variables from these files (in order of precedence):

1. Command line variables
2. .env.local (except in test mode)
3. .env.[NODE_ENV] (e.g., .env.development)
4. .env
5. wrangler.toml (for Cloudflare Workers)
Note: When NODE_ENV=test, .env.local is not loaded to ensure consistent test environments.

⚠️ Important bun pm pkg Limitations

Based on our testing, here are key limitations to be aware of:

# ❌ Version Numbers with Dots (Common Issue)
# This will FAIL:
bun pm pkg get metadata.roadmap.v2.2.0        # InvalidPath error
bun pm pkg get "metadata.roadmap[\"v2.2.0\"]" # InvalidPath error

# ✅ Workaround - Access the full object:
bun pm pkg get metadata.roadmap                # Returns complete roadmap

# ❌ Array Index Manipulation (Important Limitation)
# Setting array[index] overwrites the entire array:
bun pm pkg set "config.envValidation.required[4]"="NEW_VAR"  # ❌ Overwrites array!

# ✅ Best Practice - Set entire arrays:
bun pm pkg set "config.envValidation.required"='["VAR1", "VAR2", "VAR3"]'
🔍 Discovered Behavior: bun pm pkg set with array indices doesn't work like JavaScript arrays. It overwrites the entire array instead of extending it.
📖 Complete Details: See bun run pkg:docs for comprehensive troubleshooting and best practices!

// src/config.ts - Already implemented!
import { getConfig } from './config';

const config = getConfig();

// Fire22 API Integration Examples
const fire22Api = {
  // Get agents list
  async getAgents() {
    const response = await fetch(`${config.fire22.apiUrl}/agents`, {
      headers: {
        'Authorization': `Bearer ${config.fire22.token}`,
        'Content-Type': 'application/json'
      }
    });
    return response.json();
  },
  
  // Verify webhook signature
  verifyWebhook(body: string, signature: string): boolean {
    const expectedSignature = crypto
      .createHmac('sha256', config.fire22.webhookSecret)
      .update(body)
      .digest('hex');
    return crypto.timingSafeEqual(
      Buffer.from(signature),
      Buffer.from(expectedSignature)
    );
  }
};

🔒 Security Team & Telegram Alert Best Practices

Follow these best practices for secure and efficient security team integration and Telegram alert channels:

# Security Team Integration Best Practices
class SecurityTeamManager {
  private teamId: string;
  private token: string;
  private alertLevel: 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL';
  
  constructor(config: SecurityTeamConfig) {
    this.teamId = config.teamId;
    this.token = config.token;
    this.alertLevel = config.defaultAlertLevel;
  }
  
  async sendAlert(message: string, level: AlertLevel = this.alertLevel) {
    const alert = {
      teamId: this.teamId,
      message,
      level,
      timestamp: new Date().toISOString(),
      source: 'fire22-dashboard'
    };
    
    await this.sendToSecurityTeam(alert);
    await this.sendToTelegram(alert);
  }
  
  private async sendToSecurityTeam(alert: SecurityAlert) {
    const response = await fetch(process.env.SECURITY_MONITOR_URL, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${process.env.SECURITY_MONITOR_KEY}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(alert)
    });
    
    if (!response.ok) {
      throw new Error(`Security team alert failed: ${response.status}`);
    }
  }
  
  private async sendToTelegram(alert: SecurityAlert) {
    const botToken = process.env.SECURITY_BOT_TOKEN;
    const chatId = process.env.SECURITY_CHAT_ID;
    
    const message = this.formatTelegramMessage(alert);
    
    await fetch(`https://api.telegram.org/bot${botToken}/sendMessage`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        chat_id: chatId,
        text: message,
        parse_mode: 'HTML'
      })
    });
  }
  
  private formatTelegramMessage(alert: SecurityAlert): string {
    const emoji = this.getAlertEmoji(alert.level);
    return `${emoji} Security Alert: ${alert.level}\n\n` +
      `${alert.message}\n\n` +
      `Source: ${alert.source}\n` +
      `Time: ${alert.timestamp}`;
  }
  
  private getAlertEmoji(level: AlertLevel): string {
    switch (level) {
      case 'LOW': return '🔵';
      case 'MEDIUM': return '🟡';
      case 'HIGH': return '🟠';
      case 'CRITICAL': return '🔴';
      default: return '⚪';
    }
  }
}

# Rate Limiting Implementation
class Fire22RateLimiter {
  private requests = new Map<string, number[]>();
  private maxRequests = 1000;
  private windowMs = 60 * 1000; // 1 minute
  
  async checkLimit(identifier: string): Promise<boolean> {
    const now = Date.now();
    const userRequests = this.requests.get(identifier) || [];
    
    // Remove expired requests
    const validRequests = userRequests.filter(time => now - time < this.windowMs);
    
    if (validRequests.length >= this.maxRequests) {
      return false; // Rate limit exceeded
    }
    
    validRequests.push(now);
    this.requests.set(identifier, validRequests);
    return true;
  }
}

# Error Handling
async handleFire22Request(url: string, options: RequestInit) {
  try {
    const response = await fetch(url, options);
    
    if (response.status === 429) {
      throw new Error('Rate limit exceeded. Please retry later.');
    }
    
    if (!response.ok) {
      throw new Error(`Fire22 API error: ${response.status} ${response.statusText}`);
    }
    
    return response.json();
  } catch (error) {
    console.error('Fire22 API request failed:', error);
    throw error;
  }
}
