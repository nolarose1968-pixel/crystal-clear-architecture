# 3. Rate Limiting
const rateLimiter = new Map();
const checkRateLimit = (ip, limit = 100, window = 900000) => {
  const now = Date.now();
  const userRequests = rateLimiter.get(ip) || [];
  const validRequests = userRequests.filter(time => now - time < window);
  
  if (validRequests.length >= limit) {
    return false; // Rate limited
  }
  
  validRequests.push(now);
  rateLimiter.set(ip, validRequests);
  return true;
};

# 4. Webhook Signature Verification
const verifyWebhook = (body, signature, secret) => {
  const expectedSignature = crypto
    .createHmac('sha256', secret)
    .update(body)
    .digest('hex');
  
  return crypto.timingSafeEqual(
    Buffer.from(signature),
    Buffer.from(expectedSignature)
  );
};
🔒 Critical: Always verify webhook signatures, implement rate limiting, and rotate API keys regularly.

🔍 Data Validation Patterns
# Data Validation Helper Functions
class PermissionValidator {
  static validatePermissionData(data: any): boolean {
    if (!data || typeof data !== 'object') {
      return false;
    }
    
    // Check required fields
    const requiredFields = ['agent_id', 'permission_set'];
    for (const field of requiredFields) {
      if (!(field in data)) {
        return false;
      }
    }
    
    // Validate permission set structure
    if (!data.permission_set || typeof data.permission_set !== 'object') {
      return false;
    }
    
    // Check permission values are boolean
    for (const [permission, value] of Object.entries(data.permission_set)) {
      if (typeof value !== 'boolean') {
        return false;
      }
    }
    
    return true;
  }
  
  static validateInheritanceChain(data: any): boolean {
    if (data.inherited_from) {
      return typeof data.inherited_from === 'string' && 
             data.inherited_from.length > 0;
    }
    return true; // Optional field
  }
  
  static validateCommissionRates(data: any): boolean {
    if (data.commission_rates) {
      if (typeof data.commission_rates !== 'object') {
        return false;
      }
      
      // Check commission rates are numbers between 0 and 1
      for (const [rate, value] of Object.entries(data.commission_rates)) {
        if (typeof value !== 'number' || value < 0 || value > 1) {
          return false;
        }
      }
    }
    return true; // Optional field
  }
}
🚨 Error Handling for Data Responses
# Comprehensive Error Handling
async function handlePermissionResponse(response: any) {
  try {
    if (!response.success) {
      const errorCode = response.error.code;
      
      switch (errorCode) {
        case '1006': // Permission Validation Failed
          await handlePermissionValidationError(response);
          break;
        case '1007': // Invalid Permission Format
          await handlePermissionFormatError(response);
          break;
        case '1008': // Permission Inheritance Error
          await handleInheritanceError(response);
          break;
        default:
          handleGenericError(response.error);
      }
      return;
    }
    
    // Validate successful response data
    if (response.data) {
      if (!PermissionValidator.validatePermissionData(response.data.permissions)) {
        throw new Error('Invalid permission data structure');
      }
      
      if (!PermissionValidator.validateInheritanceChain(response.data.permissions)) {
        throw new Error('Invalid inheritance chain');
      }
      
      if (!PermissionValidator.validateCommissionRates(response.data.permissions)) {
        throw new Error('Invalid commission rates');
      }
      
      // Process valid permission data
      await processValidPermissions(response.data);
    }
    
  } catch (error) {
    console.error('Error handling permission response:', error);
    await handleUnexpectedError(error);
  }
}

async function handlePermissionValidationError(response: any) {
  const { validationErrors, affectedAgents, permissionMatrix } = response.data;
  
  console.error('Permission validation failed:', {
    errors: validationErrors,
    agents: affectedAgents,
    matrix: permissionMatrix
  });
  
  // Log validation errors for debugging
  validationErrors.forEach((error: any) => {
    console.error(`${error.severity.toUpperCase()}: ${error.field} - ${error.issue}`);
  });
  
  // Show user-friendly error message
  showUserError('Permission validation failed. Please check agent configuration.');
  
  // Attempt automatic repair if possible
  if (permissionMatrix.healthScore > 70) {
    await attemptPermissionRepair(affectedAgents);
  }
}

🔄 Implement Retry Logic

Use exponential backoff for transient errors like rate limits and timeouts.

async function retryWithBackoff(apiCall, maxRetries = 3) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await apiCall();
    } catch (error) {
      if (attempt === maxRetries) throw error;
      
      const delay = Math.pow(2, attempt) * 1000;
      await sleep(delay);
    }
  }
}

# Circuit Breaker Implementation
class CircuitBreaker {
  constructor(failureThreshold = 5, timeout = 60000) {
    this.failureThreshold = failureThreshold;
    this.timeout = timeout;
    this.failures = 0;
    this.lastFailureTime = 0;
    this.state = 'CLOSED'; // CLOSED, OPEN, HALF_OPEN
  }
  
  async call(apiCall) {
    if (this.state === 'OPEN') {
      if (Date.now() - this.lastFailureTime > this.timeout) {
        this.state = 'HALF_OPEN';
      } else {
        throw new Error('Circuit breaker is OPEN');
      }
    }
    
    try {
      const result = await apiCall();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }
  
  onSuccess() {
    this.failures = 0;
    this.state = 'CLOSED';
  }
  
  onFailure() {
    this.failures++;
    this.lastFailureTime = Date.now();
    
    if (this.failures >= this.failureThreshold) {
      this.state = 'OPEN';
    }
  }
}
